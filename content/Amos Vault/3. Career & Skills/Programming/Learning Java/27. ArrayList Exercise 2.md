QUESTION :
Create a menu Programme with the following Options.
1. Add Element .
2. Remove Element .
3. Display Element .
4. Exit .
This Progran runs infinitely, until the user chooses the fourth Option.

SOULTION
```java
public static void main(String[] args) {  
    ArrayList<Integer> numbers = new ArrayList<>();  
    Scanner input = new Scanner(System.in);  
    
     while (true){   //Use the infinite loop
     //Call the method to display the menu
         printMenu();  
         
         //Take the input from the user
         System.out.print("Enter your choice : ");  
         int choice = input.nextInt();  
  
         if(choice==1){  
             System.out.print("Enter Element to Add : ");  
             numbers.add(input.nextInt());  
             System.out.println("Added");  
         } else if (choice==2) {  
             System.out.print("Enter element to remove : ");  
             int elementToRemove = input.nextInt();  
             if (numbers.contains(elementToRemove)){  
                 numbers.remove(Integer.valueOf(elementToRemove));  
                 System.out.println("Removed.");  
             }  
             else{  
                 System.out.println("Element not found!");  
             }  
         } else if (choice==3) {  
             System.out.println("Your List : "+numbers);  
         }  
         else if (choice==4){  
             System.out.println("Good bye!");  
             break;  
         }  
     }  
}  
  
  //Create the method to print the Menu
public static void  printMenu(){  
    System.out.println();  
    System.out.println("1. Add Element");  
    System.out.println("2. Remove Element");  
    System.out.println("3. Display Element");  
    System.out.println("4. Exit");  
    System.out.println();  
}
```
